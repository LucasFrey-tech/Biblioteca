// Users Table
Table users {
  id_user SERIAL [pk]
  name VARCHAR
  email VARCHAR
  password VARCHAR
  role VARCHAR
  registration_date TIMESTAMP
  status VARCHAR
}

// Books Table
Table books {
  id_book SERIAL [pk]
  title VARCHAR
  author VARCHAR
  genre VARCHAR
  description TEXT
  type VARCHAR
  price DECIMAL
  stock INTEGER
  virtual_url VARCHAR
}

// Orders Table
Table orders {
  id_order SERIAL [pk]
  id_user INTEGER [ref: > users.id_user]
  order_date TIMESTAMP
  status VARCHAR
  total DECIMAL
}

// Reservations Table
Table reservations {
  id_reservation SERIAL [pk]
  id_user INTEGER [ref: > users.id_user]
  id_book INTEGER [ref: > books.id_book]
  reservation_date TIMESTAMP
  status VARCHAR
}

// Shopping Cart Table
Table shopping_cart {
  id_cart SERIAL [pk]
  id_user INTEGER [ref: > users.id_user]
  creation_date TIMESTAMP
}

// Reviews and Ratings Table
Table reviews {
  id_review SERIAL [pk]
  id_user INTEGER [ref: > users.id_user]
  id_book INTEGER [ref: > books.id_book]
  rating INTEGER
  comment TEXT
  review_date TIMESTAMP
}

// Comments Table
Table comments {
  id_comment SERIAL [pk]
  id_user INTEGER [ref: > users.id_user]
  id_book INTEGER [ref: > books.id_book]
  comment TEXT
  comment_date TIMESTAMP
}

// Subscription Plans Table
Table subscription_plans {
  id_subscription SERIAL [pk]
  id_user INTER [ref: > users.id_user]
  name VARCHAR
  allowed_books INTEGER
  time_period VARCHAR
  price DECIMAL
}
